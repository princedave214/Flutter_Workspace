3. POP vs OOP Comparison Program
Procedural Programming (POP)
#include <iostream>
using namespace std;

int main()
{
    int length, width;
    cout << "Enter length and width: ";
    cin >> length >> width;

    int area = length * width;
    cout << "Area of Rectangle = " << area;
    return 0;
}

Object-Oriented Programming (OOP)
#include <iostream>
using namespace std;

class Rectangle
{
public:
    int length, width;
    int area()
    {
        return length * width;
    }
};

int main()
{
    Rectangle r;
    cout << "Enter length and width: ";
    cin >> r.length >> r.width;

    cout << "Area of Rectangle = " << r.area();
    return 0;
}


Objective:
Shows the difference between POP (function-based) and OOP (class-based).

2. Number Guessing Game
#include <iostream>
using namespace std;

int main()
{
    int num = 50, guess;
    do
    {
        cin >> guess;
        if (guess > num) cout << "Too High\n";
        else if (guess < num) cout << "Too Low\n";
    } while (guess != num);

    cout << "Correct!";
    return 0;
}

Matrix Addition
#include <iostream>
using namespace std;

int main()
{
    int a[2][2], b[2][2], c[2][2];
    for(int i=0;i<2;i++)
        for(int j=0;j<2;j++)
            cin>>a[i][j]>>b[i][j];

    for(int i=0;i<2;i++)
        for(int j=0;j<2;j++)
            c[i][j]=a[i][j]+b[i][j];

    return 0;
}

Palindrome Check
#include <iostream>
#include <string>
using namespace std;

int main()
{
    string s;
    cin >> s;
    string r = string(s.rbegin(), s.rend());

    if (s == r) cout << "Palindrome";
    else cout << "Not Palindrome";
    return 0;
}
